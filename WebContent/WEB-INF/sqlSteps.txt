create database companyDB using below 2 steps

create database CompanyDB;
use CompanyDB;

create companies table:

CREATE TABLE CompanyDB.`companies` (
  `CORPORATE_IDENTIFICATION_NUMBER` varchar(25) NOT NULL,
  `DATE_OF_REGISTRATION` date DEFAULT NULL,
  `COMPANY_NAME` varchar(45) DEFAULT NULL,
  `COMPANY_STATUS` varchar(45) DEFAULT NULL,
  `COMPANY_CLASS` varchar(10) DEFAULT NULL,
  `COMPANY_CATEGORY` varchar(45) DEFAULT NULL,
  `AUTHORIZED_CAPITAL` varchar(45) DEFAULT NULL,
  `PAIDUP_CAPITAL` varchar(45) DEFAULT NULL,
  `REGISTERED_STATE` varchar(45) DEFAULT NULL,
  `REGISTRAR_OF_COMPANIES` varchar(45) DEFAULT NULL,
  `PRINCIPAL_BUSINESS_ACTIVITY` varchar(100) DEFAULT NULL,
  `REGISTERED_OFFICE_ADDRESS` varchar(100) DEFAULT NULL,
  `SUB_CATEGORY` varchar(100) DEFAULT NULL,
  PRIMARY KEY (`CORPORATE_IDENTIFICATION_NUMBER`)
)
